514  private Cell detectAndAddHit(float x, float y) {
515  final Cell cell = checkForNewHit(x, y);
516  if (cell != null) {
517 
518  // check for gaps in existing pattern
519  Cell fillInGapCell = null;
520  final ArrayList<Cell> pattern = mPattern;
521  if (!pattern.isEmpty()) {
522  final Cell lastCell = pattern.get(pattern.size() - 1);
523  int dRow = cell.row - lastCell.row;
524  int dColumn = cell.column - lastCell.column;
525 
526  int fillInRow = lastCell.row;
527  int fillInColumn = lastCell.column;
528 
529  if (Math.abs(dRow) == 2 && Math.abs(dColumn) != 1) {
530  fillInRow = lastCell.row + ((dRow > 0) ? 1 : -1);
531  }
532 
533  if (Math.abs(dColumn) == 2 && Math.abs(dRow) != 1) {
534  fillInColumn = lastCell.column + ((dColumn > 0) ? 1 : -1);
535  }
536 
537  fillInGapCell = Cell.of(fillInRow, fillInColumn);
538  }
539 
540  if (fillInGapCell != null &&
541  !mPatternDrawLookup[fillInGapCell.row][fillInGapCell.column]) {
542  addCellToPattern(fillInGapCell);
543  }
544  addCellToPattern(cell);
545  if (mEnableHapticFeedback) {
546  performHapticFeedback(HapticFeedbackConstants.VIRTUAL_KEY,
547  HapticFeedbackConstants.FLAG_IGNORE_VIEW_SETTING
548  | HapticFeedbackConstants.FLAG_IGNORE_GLOBAL_SETTING);
549  }
550  return cell;
551  }
552  return null;
553  }

672  private void handleActionMove(MotionEvent event) {
673  // Handle all recent motion events so we don't skip any cells even when the device
674  // is busy...
675  final int historySize = event.getHistorySize();
676  for (int i = 0; i < historySize + 1; i++) {
677  final float x = i < historySize ? event.getHistoricalX(i) : event.getX();
678  final float y = i < historySize ? event.getHistoricalY(i) : event.getY();
679  final int patternSizePreHitDetect = mPattern.size();
680  Cell hitCell = detectAndAddHit(x, y);
681  final int patternSize = mPattern.size();
682  if (hitCell != null && patternSize == 1) {
683  mPatternInProgress = true;
684  notifyPatternStarted();
685  }
686  // note current x and y for rubber banding of in progress patterns
687  final float dx = Math.abs(x - mInProgressX);
688  final float dy = Math.abs(y - mInProgressY);
689  if (dx + dy > mSquareWidth * 0.01f) {
690  float oldX = mInProgressX;
691  float oldY = mInProgressY;
692 
693  mInProgressX = x;
694  mInProgressY = y;
695 
696  if (mPatternInProgress && patternSize > 0) {
697  final ArrayList<Cell> pattern = mPattern;
698  final float radius = mSquareWidth * mDiameterFactor * 0.5f;
699 
700  final Cell lastCell = pattern.get(patternSize - 1);
701 
702  float startX = getCenterXForColumn(lastCell.column);
703  float startY = getCenterYForRow(lastCell.row);
704 
705  float left;
706  float top;
707  float right;
708  float bottom;
709 
710  final Rect invalidateRect = mInvalidate;
711 
712  if (startX < x) {
713  left = startX;
714  right = x;
715  } else {
716  left = x;
717  right = startX;
718  }
719 
720  if (startY < y) {
721  top = startY;
722  bottom = y;
723  } else {
724  top = y;
725  bottom = startY;
726  }
727 
728  // Invalidate between the pattern's last cell and the current location
729  invalidateRect.set((int) (left - radius), (int) (top - radius),
730  (int) (right + radius), (int) (bottom + radius));
731 
732  if (startX < oldX) {
733  left = startX;
734  right = oldX;
735  } else {
736  left = oldX;
737  right = startX;
738  }
739 
740  if (startY < oldY) {
741  top = startY;
742  bottom = oldY;
743  } else {
744  top = oldY;
745  bottom = startY;
746  }
747 
748  // Invalidate between the pattern's last cell and the previous location
749  invalidateRect.union((int) (left - radius), (int) (top - radius),
750  (int) (right + radius), (int) (bottom + radius));
751 
752  // Invalidate between the pattern's new cell and the pattern's previous cell
753  if (hitCell != null) {
754  startX = getCenterXForColumn(hitCell.column);
755  startY = getCenterYForRow(hitCell.row);
756 
757  if (patternSize >= 2) {
758  // (re-using hitcell for old cell)
759  hitCell = pattern.get(patternSize - 1 - (patternSize - patternSizePreHitDetect));
760  oldX = getCenterXForColumn(hitCell.column);
761  oldY = getCenterYForRow(hitCell.row);
762 
763  if (startX < oldX) {
764  left = startX;
765  right = oldX;
766  } else {
767  left = oldX;
768  right = startX;
769  }
770 
771  if (startY < oldY) {
772  top = startY;
773  bottom = oldY;
774  } else {
775  top = oldY;
776  bottom = startY;
777  }
778  } else {
779  left = right = startX;
780  top = bottom = startY;
781  }
782 
783  final float widthOffset = mSquareWidth / 2f;
784  final float heightOffset = mSquareHeight / 2f;
785 
786  invalidateRect.set((int) (left - widthOffset),
787  (int) (top - heightOffset), (int) (right + widthOffset),
788  (int) (bottom + heightOffset));
789  }
790 
791  invalidate(invalidateRect);
792  } else {
793  invalidate();
794  }
795  }
796  }

820  private void handleActionDown(MotionEvent event) {
821  resetPattern();
822  final float x = event.getX();
823  final float y = event.getY();
824  final Cell hitCell = detectAndAddHit(x, y);
825  if (hitCell != null) {
826  mPatternInProgress = true;
827  mPatternDisplayMode = DisplayMode.Correct;
828  notifyPatternStarted();
829  } else {
830  mPatternInProgress = false;
831  notifyPatternCleared();
832  }
833  if (hitCell != null) {
834  final float startX = getCenterXForColumn(hitCell.column);
835  final float startY = getCenterYForRow(hitCell.row);
836 
837  final float widthOffset = mSquareWidth / 2f;
838  final float heightOffset = mSquareHeight / 2f;
839 
840  invalidate((int) (startX - widthOffset), (int) (startY - heightOffset),
841  (int) (startX + widthOffset), (int) (startY + heightOffset));
842  }
843  mInProgressX = x;
844  mInProgressY = y;
845  if (PROFILE_DRAWING) {
846  if (!mDrawingProfilingStarted) {
847  Debug.startMethodTracing("LockPatternDrawing");
848  mDrawingProfilingStarted = true;
849  }

993  private void drawArrow(Canvas canvas, float leftX, float topY, Cell start, Cell end) {
994  boolean green = mPatternDisplayMode != DisplayMode.Wrong;
995 
996  final int endRow = end.row;
997  final int startRow = start.row;
998  final int endColumn = end.column;
999  final int startColumn = start.column;
1000 
1001  // offsets for centering the bitmap in the cell
1002  final int offsetX = ((int) mSquareWidth - mBitmapWidth) / 2;
1003  final int offsetY = ((int) mSquareHeight - mBitmapHeight) / 2;
1004 
1005 // Log.v("--->>", offsetX+" "+offsetY+" offset");
1006 
1007  // compute transform to place arrow bitmaps at correct angle inside circle.
1008  // This assumes that the arrow image is drawn at 12:00 with it's top edge
1009  // coincident with the circle bitmap's top edge.
1010  Bitmap arrow = green ? mBitmapArrowGreenUp : mBitmapArrowRedUp;
1011  final int cellWidth = mBitmapWidth;
1012  final int cellHeight = mBitmapHeight;
1013 
1014  // the up arrow bitmap is at 12:00, so find the rotation from x axis and add 90 degrees.
1015  final float theta = (float) Math.atan2(
1016  (double) (endRow - startRow), (double) (endColumn - startColumn));
1017  final float angle = (float) Math.toDegrees(theta) + 90.0f;
1018 
1019  // compose matrix
1020  float sx = Math.min(mSquareWidth / mBitmapWidth, 1.0f);
1021  float sy = Math.min(mSquareHeight / mBitmapHeight, 1.0f);
1022  mArrowMatrix.setTranslate(leftX + offsetX, topY + offsetY); // transform to cell position
1023  mArrowMatrix.preTranslate(mBitmapWidth/2, mBitmapHeight/2);
1024  mArrowMatrix.preScale(sx, sy);
1025  mArrowMatrix.preTranslate(-mBitmapWidth/2, -mBitmapHeight/2);
1026  mArrowMatrix.preRotate(angle, cellWidth / 2.0f, cellHeight / 2.0f); // rotate about cell center
1027  mArrowMatrix.preTranslate((cellWidth - arrow.getWidth()) / 2.0f, 0.0f); // translate to 12:00 pos
1028  canvas.drawBitmap(arrow, mArrowMatrix, mPaint);
1029  }
